Create Web Directory

gopala@gopala-VMware20-1:~$ cd /var/www
sudo mkdir tutorial
cd tutorial
sudo "${EDITOR:-vi}" index.html

Configure sites-enabled

gopala@gopala-VMware20-1:/etc/nginx/sites-enabled$ echo "server { listen 81; listen [::]:81; server_name example.ubuntu.com; root /var/www/tutorial; index index.html; location / { try_files \$uri \$uri/ =404; } }" | sudo tee /etc/nginx/sites-available/tutorial && sudo ln -s /etc/nginx/sites-available/tutorial /etc/nginx/sites-enabled/ && sudo nginx -t && sudo systemctl restart nginx
